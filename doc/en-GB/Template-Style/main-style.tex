\newcommand{\myPrintChapterLabel}[1]{
	\ifthenelse{\equal{#1}{A}}
		{\myAppendixLabel} 
		{\ifthenelse{\equal{#1}{B}}
			{\myAppendixLabel} 
			{\ifthenelse{\equal{#1}{C}}
				{\myAppendixLabel} 
				{\ifthenelse{\equal{#1}{D}}
					{\myAppendixLabel} 
					{\ifthenelse{\equal{#1}{E}}
						{\myAppendixLabel} 
						{\ifthenelse{\equal{#1}{F}}
							{\myAppendixLabel} 
							{\ifthenelse{\equal{#1}{G}}
								{\myAppendixLabel} 
								{\myChapterLabel}}}}}}} 
}
